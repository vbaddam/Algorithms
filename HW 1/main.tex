\textbf{Q5.}\\
Min-Heapify and Max-Heapify has the same time complexity of \mathcal{O}(\log n)

\begin{algorithm}
\caption{Min-Heapify}\label{euclid}
\begin{algorithmic}[1]
\Procedure{\textsc{Min-Heapify}(\textit{A,i})}{}
\State $l$ = \textsc{Left}(\textit{i})
\State \textit{r} = \textsc{Left}(\textit{i})
\State \textit{n} = \textit{A.heap-size}
\If {$l \leq n $ and $A[l] < A[i]$} $smallest = l$
\EndIf
\State \textbf{else} $smallest = i$
\If {$r \leq n $ and $A[r] < A[i]$} $smallest = r$
\EndIf
\If {$smallest \neq i$}
\State Swap $A[i]$ and $A[smallest]$
\State \textsc{Min-Heapify}(\textit{A,i})
\EndIf
\EndProcedure
\end{algorithmic}
\end{algorithm}
